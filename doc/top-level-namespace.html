get('/profile/:id') do
    result = profile(params)    
    if result.key?(:user)
        session[:error] = ""
        slim(:user, locals:{posts: result[:posts], session: session, user: result[:user]})
    else
        session[:error] = result[:error]
        redirect('/')
    end
end
